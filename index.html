                    rankClass = 'rank-4';

                    rankIcon = '🎖️';

                } else {

                    rankIcon = rank;

                }

                

                html += `

                    <div class="leaderboard-row">

                        <div class="rank-badge ${rankClass}">${rankIcon}</div>

                        <div class="player-details">

                            <div class="player-name">${player.name}</div>

                            <div class="player-captain">⚡ ${player.captain}</div>

                        </div>

                        <div class="score-prize">

                            <div class="score">${player.score}</div>

                            ${player.prize > 0 ? `<div class="prize">+₹${player.prize}</div>` : ''}

                        </div>

                    </div>

                `;

            });

            

            document.getElementById('leaderboard').innerHTML = html;

        }


        function updateOverallStandings() {

            // Calculate total earnings for each player (OVERALL, not per GW)

            const earnings = {};

            

            // Get all unique players

            const allPlayers = new Set();

            Object.values(gameData).forEach(gw => {

                gw.players.forEach(player => allPlayers.add(player));

            });

            

            // Initialize

            allPlayers.forEach(player => {

                earnings[player] = { 

                    grossWon: 0, 

                    gamesPlayed: 0,

                    lockin: 0,

                    net: 0,

                    wins: [],

                    positions: []

                };

            });

            

            // Calculate earnings and games played

            Object.entries(gameData).forEach(([gwNum, gwData]) => {

                if (gwData.results.length > 0) {

                    gwData.results.forEach((result, index) => {

                        if (gwData.players.includes(result.name)) {

                            earnings[result.name].gamesPlayed++;

                            earnings[result.name].lockin += 100; // Each game costs ₹100

                            

                            if (result.prize > 0) {

                                earnings[result.name].grossWon += result.prize;

                                const position = index + 1;

                                earnings[result.name].wins.push({ 

                                    gw: parseInt(gwNum), 

                                    position,

                                    amount: result.prize 

                                });

                            }

                            

                            earnings[result.name].positions.push({

                                gw: parseInt(gwNum),

                                position: index + 1

                            });

                        }

                    });

                }

            });

            

            // Calculate net earnings

            Object.keys(earnings).forEach(player => {

                earnings[player].net = earnings[player].grossWon - earnings[player].lockin;

            });

            

            // Sort by net earnings

            const sorted = Object.entries(earnings)

                .filter(([_, data]) => data.gamesPlayed > 0)

                .sort((a, b) => b[1].net - a[1].net);

            

            let html = '';

            sorted.forEach(([name, data], index) => {

                const netClass = data.net > 0 ? 'net-positive' : data.net < 0 ? 'net-negative' : 'net-neutral';

                

                // Create badges for wins

                const badges = data.wins.map(w => {

                    const medals = ['🥇', '🥈', '🥉', '🎖️'];

                    let badgeClass = '';

                    if (w.position === 1) badgeClass = 'gold';

                    else if (w.position === 2) badgeClass = 'silver';

                    else if (w.position === 3) badgeClass = 'bronze';

                    

                    return `<span class="badge ${badgeClass}">GW${w.gw} ${medals[w.position - 1]}</span>`;

                }).join('');

                

                html += `

                    <div class="overall-card" style="animation-delay: ${index * 0.05}s">

                        <div class="overall-header">

                            <div class="overall-player">

                                <div class="overall-name">${name}</div>

                                <div class="overall-stats">

                                    <span class="stat-item">🎮 ${data.gamesPlayed} games</span>

                                    <span class="stat-item">💰 ₹${data.lockin} locked</span>

                                </div>

                            </div>

                            <div class="overall-earnings">

                                <div class="net-value ${netClass}">

                                    ${data.net >= 0 ? '+' : ''}₹${Math.abs(data.net)}

                                </div>

                                <div class="gross-value">Won: ₹${data.grossWon}</div>

                            </div>

                        </div>

                        <div class="badge-container">

                            ${badges || '<span style="color: #718096; font-size: 0.85rem;">No wins yet</span>'}

                        </div>

                    </div>

                `;

            });

            

            document.getElementById('overallStandings').innerHTML = html;

        }


        function updateStats() {

            // Calculate OVERALL stats across all GWs

            const playerStats = {};

            

            // Initialize all players

            const allPlayers = new Set();

            Object.values(gameData).forEach(gw => {

                gw.players.forEach(player => allPlayers.add(player));

            });

            

            allPlayers.forEach(player => {

                playerStats[player] = {

                    captainPoints: 0,

                    totalScore: 0,

                    gamesPlayed: 0,

                    wins: 0,

                    highestScore: 0,

                    lowestScore: 999,

                    avgScore: 0,

                    netEarnings: 0

                };

            });

            

            // Calculate stats

            Object.values(gameData).forEach(gwData => {

                if (gwData.results.length > 0) {

                    gwData.results.forEach((result, index) => {

                        if (gwData.players.includes(result.name)) {

                            const stats = playerStats[result.name];

                            stats.gamesPlayed++;

                            stats.totalScore += result.score;

                            stats.highestScore = Math.max(stats.highestScore, result.score);

                            stats.lowestScore = Math.min(stats.lowestScore, result.score);

                            

                            // Estimate captain points (assuming captain scored ~30% of total)

                            stats.captainPoints += Math.round(result.score * 0.3);

                            

                            if (result.prize > 0) {

                                stats.wins++;

                                stats.netEarnings += result.prize;

                            }

                            stats.netEarnings -= 100; // Deduct entry fee

                        }

                    });

                }

            });

            

            // Calculate averages

            Object.keys(playerStats).forEach(player => {

                const stats = playerStats[player];

                if (stats.gamesPlayed > 0) {

                    stats.avgScore = Math.round(stats.totalScore / stats.gamesPlayed);

                }

            });

            

            // Top Captain Points

            const topCaptains = Object.entries(playerStats)

                .filter(([_, stats]) => stats.gamesPlayed > 0)

                .sort((a, b) => b[1].captainPoints - a[1].captainPoints)

                .slice(0, 5);

            

            const captainHTML = topCaptains.map(([name, stats]) => `

                <div class="player-stat">

                    <div class="player-stat-name">${name}</div>

                    <div class="player-stat-value">${stats.captainPoints} pts</div>

                </div>

            `).join('');

            document.getElementById('captainStats').innerHTML = captainHTML;

            

            // Fun Awards

            const funAwards = [

                {

                    emoji: '👑',

                    title: 'Money King',

                    winner: Object.entries(playerStats).sort((a, b) => b[1].netEarnings - a[1].netEarnings)[0],

                    desc: 'Highest net earnings across all gameweeks'

                },

                {

                    emoji: '🎯',

                    title: 'Captain Fantastic',

                    winner: Object.entries(playerStats).sort((a, b) => b[1].captainPoints - a[1].captainPoints)[0],

                    desc: 'Most points from captain choices'

                },

                {

                    emoji: '🚀',

                    title: 'Single GW Legend',

                    winner: Object.entries(playerStats).sort((a, b) => b[1].highestScore - a[1].highestScore)[0],

                    desc: 'Highest score in any single gameweek'

                },

                {

                    emoji: '😴',

                    title: 'Sleepy Manager',

                    winner: Object.entries(playerStats)

                        .filter(([_, stats]) => stats.gamesPlayed > 0)

                        .sort((a, b) => a[1].lowestScore - b[1].lowestScore)[0],

                    desc: 'Lowest score in any gameweek (still playing)'

                },

                {

                    emoji: '🎢',

                    title: 'Mr. Consistent',

                    winner: Object.entries(playerStats)

                        .filter(([_, stats]) => stats.gamesPlayed > 0)

                        .sort((a, b) => b[1].avgScore - a[1].avgScore)[0],

                    desc: 'Highest average score per gameweek'

                }

            ];

            

            const funHTML = funAwards.map(award => {

                if (!award.winner) return '';

                const [name, stats] = award.winner;

                let value = '';

                if (award.title === 'Money King') value = `₹${stats.netEarnings}`;

                else if (award.title === 'Captain Fantastic') value = `${stats.captainPoints} pts`;

                else if (award.title === 'Single GW Legend') value = `${stats.highestScore} pts`;

                else if (award.title === 'Sleepy Manager') value = `${stats.lowestScore} pts`;

                else if (award.title === 'Mr. Consistent') value = `${stats.avgScore} avg`;

                

                return `

                    <div class="fun-award">

                        <div class="fun-award-header">

                            <div class="fun-award-title">

                                <span>${award.emoji}</span>

                                <span>${award.title}</span>

                            </div>

                            <div class="fun-award-winner">${name}</div>

                        </div>

                        <div class="fun-award-desc">${award.desc} - ${value}</div>

                    </div>

                `;

            }).join('');

            document.getElementById('funStats').innerHTML = funHTML;

            

            // Earnings Chart

            const topEarners = Object.entries(playerStats)

                .filter(([_, stats]) => stats.gamesPlayed > 0)

                .sort((a, b) => b[1].netEarnings - a[1].netEarnings)

                .slice(0, 8);

            

            const maxEarning = Math.max(...topEarners.map(([_, stats]) => Math.abs(stats.netEarnings)));

            

            const chartHTML = topEarners.map(([name, stats], index) => {

                const height = Math.abs(stats.netEarnings) / maxEarning * 120;

                const isPositive = stats.netEarnings >= 0;

                

                return `

                    <div class="chart-bar">

                        <div class="bar" style="

                            height: ${height}px; 

                            background: ${isPositive 

                                ? 'linear-gradient(to top, #48bb78, rgba(72, 187, 120, 0.5))' 

                                : 'linear-gradient(to top, #e53e3e, rgba(229, 62, 62, 0.5))'};

                            animation-delay: ${index * 0.1}s;

                        ">

                            <div class="bar-value">${stats.netEarnings >= 0 ? '+' : ''}₹${Math.abs(stats.netEarnings)}</div>

                        </div>

                        <div class="bar-label">${name.substring(0, 6)}</div>

                    </div>

                `;

            }).join('');

            document.getElementById('earningsChart').innerHTML = chartHTML;

        }
